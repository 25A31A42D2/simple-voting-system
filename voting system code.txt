#include <stdio.h>

int main() {
int choice;
int votesA = 0, votesB = 0, votesC = 0;

printf("=== Simple Voting System ===\n");
printf("1. Candidate A\n");
printf("2. Candidate B\n");
printf("3. Candidate C\n");
printf("4. Exit & Show Results\n");

while (1) {
printf("\nEnter your vote (1-4): ");
scanf("%d", &choice);

switch (choice) {
case 1:
votesA++;
printf("You voted for Candidate A\n");
break;
case 2:
votesB++;
printf("You voted for Candidate B\n");
break;
case 3:
votesC++;
printf("You voted for Candidate C\n");
break;
case 4:
printf("\n=== Voting Results ===\n");
printf("Candidate A: %d votes\n", votesA);
printf("Candidate B: %d votes\n", votesB);
printf("Candidate C: %d votes\n", votesC);

if (votesA > votesB && votesA > votesC)
printf("Winner: Candidate A\n");
else if (votesB > votesA && votesB > votesC)
printf("Winner: Candidate B\n");
else if (votesC > votesA && votesC > votesB)
printf("Winner: Candidate C\n");
else
printf("It's a tie!\n");

return 0;

default:
printf("Invalid choice! Try again.\n");
}
}

return 0;
}
